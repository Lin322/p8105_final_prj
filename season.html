test = injuries %>% 
  select(code = prod1, everything())

new = merge(test, products, by = "code")
season.df = 
  new %>% 
  separate(trmt_date, c("year","month","day"), sep = "-") %>% 
  mutate(
    month = as.double(month),
      season = case_when(
      month %in% 9:11 ~ "Fall",
      month %in% 3:5  ~ "Spring",
      month %in% 6:8  ~ "Summer",
      TRUE ~ "Winter"))
